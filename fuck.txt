

function bruh {
    $1 = addentity;
    $2 = addsprite $1 1;
    setbounds $2 0 165 50 150 100;
    setuv $2 0 0 0 1 1;
    settex $2 0;

    $3 = addsprite $1 1;
    setbounds $3 0 115 150 50 50;
    setuv $3 0 0 0 1 1;
    settex $3 0;
   
    addcomponent $1 physics;
    setattr $1 physics.accel 1 1;
    setattr $1 physics.velocity_cap 5 5;

    moveto $2   90 90;
    moveto $3   90 180;
}
